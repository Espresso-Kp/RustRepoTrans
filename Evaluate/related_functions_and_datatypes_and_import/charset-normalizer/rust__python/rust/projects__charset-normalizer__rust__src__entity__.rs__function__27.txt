pub fn decoded_payload(&self) -> Option<&str> {
        self.decoded_payload.as_deref()
    }

pub fn add_submatch(&mut self, submatch: &CharsetMatch) {
        self.submatch.push(submatch.clone());
        //self.decoded_payload = None;
    }

pub fn len(&self) -> usize {
        self.items.len()
    }
