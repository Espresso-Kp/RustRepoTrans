pub(crate) fn validate_namespace(namespace: &NamespaceIdent) -> Result<String> {
    let name = namespace.as_ref();

    if name.len() != 1 {
        return Err(Error::new(
            ErrorKind::DataInvalid,
            format!(
                "Invalid database name: {:?}, hierarchical namespaces are not supported",
                namespace
            ),
        ));
    }

    let name = name[0].clone();

    if name.is_empty() {
        return Err(Error::new(
            ErrorKind::DataInvalid,
            "Invalid database, provided namespace is empty.",
        ));
    }

    Ok(name)
}

    async fn drop_namespace(&self, namespace: &NamespaceIdent) -> Result<()> {
        let db_name = validate_namespace(namespace)?;

        let table_list = self.list_tables(namespace).await?;

        // Check if the namespace (database) is not empty
        if !table_list.is_empty() {
            return Err(NamespaceNotEmptyError(format!("Database {} is not empty", db_name)).into());

        match self.client.0.delete_database(delete_database_input).await {
            Ok(_) => Ok(()),

                SdkError::ServiceError(service_error) => {
                    let error = service_error.into_err();
                    if error.is_database_not_found_exception() || error.is_entity_not_found_exception() {
                        Err(NoSuchNamespaceError(format!("Database does not exist: {}", db_name)).into())

                        Err(from_aws_sdk_error(SdkError::ServiceError(service_error)))
                    }
                },
                _ => Err(from_aws_sdk_error(err)),
            }
        }
    }
}
