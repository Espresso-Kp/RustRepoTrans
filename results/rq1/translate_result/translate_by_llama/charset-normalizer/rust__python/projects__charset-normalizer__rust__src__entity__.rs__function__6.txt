pub fn multi_byte_usage(&self) -> f32 {
        let decoded_chars = self.decoded_payload().unwrap_or_default().chars().count() as f32;
        let payload_len = self.payload.len() as f32;

        1.0 - (decoded_chars / payload_len)
    }
