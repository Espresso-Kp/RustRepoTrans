pub fn kdf2(sha: usize, z: &[u8], p: Option<&[u8]>, olen: usize, k: &mut [u8]) {
    let h_len = sha;
    let mut key = vec![0u8; olen];
    let mut h = vec![0u8; h_len];

    for counter in 1..=((olen + h_len - 1) / h_len) {
        hashit(sha, z, Some(z), counter, &mut h, 0);
        let available_space = olen - key.len();
        if available_space >= h_len {
            key.extend_from_slice(&h);
        } else {
            let slice_len = if available_space == 0 { h_len } else { available_space };
            key.extend_from_slice(&h[..slice_len]);
        }
    }

    let key_len = key.len();
    for i in 0..olen {
        k[i] = key[i % key_len];
    }
}
