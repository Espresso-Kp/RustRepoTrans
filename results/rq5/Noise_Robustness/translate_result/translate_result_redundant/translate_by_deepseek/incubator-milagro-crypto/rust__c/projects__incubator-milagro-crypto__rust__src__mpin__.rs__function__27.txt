        }
        didit = true;

        }
        h.process_array(id);
        let hs = h.hash();

        }
        didit = true;
    }
    if sha == SHA512 {
        let mut h = HASH512::new();
        if n > 0 {
            h.process_num(n as i32)
        }

            r[i] = hs[i];

        didit = true;
    }

        return false;
